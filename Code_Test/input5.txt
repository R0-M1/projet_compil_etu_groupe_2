int add(int a, int b) {
    return a + b;
}

bool isGreater(int a, int b) {
    return a > b;
}

int main() {
    int x = 10;
    int y = 20;
    bool z = x < y;

    print(x);
    print(z);

    int result = add(x, y);
    print(result);

    if (isGreater(result, 25)) {
        print(x);
    } else {
        print(y);
    }

    int[] array = {1, 2, 3};
    int tmp = array[0];
    print(tmp);

    return 0;
}
